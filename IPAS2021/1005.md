### C



```c
#include <stdio.h>

int main() {
    // 宣告變數 single_char 為 char 整數型態，單引號將字元包起來
    char single_char = 'C';

    // %d 為整數格式，印出字元碼。\n 為換行符號
    printf("%d\n", single_char);
    // %d 為字元格式，印出字元
    printf("%c\n", single_char);
}
```
